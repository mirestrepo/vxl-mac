# ./v3p/png/CMakeListsLink.txt

LINK_LIBRARIES( png )

INCLUDE_DIRECTORIES( ${allvxl_SOURCE_DIR}/v3p/png )

IF (LIBRARY_OUTPUT_PATH MATCHES '^$')
  LINK_DIRECTORIES( ${allvxl_BINARY_DIR}/v3p/png )
ENDIF (LIBRARY_OUTPUT_PATH MATCHES '^$')

IF (CYGWIN)
  # in Cygwin a define is needed by any file including vxl/v3p/png/png.h
  IF(BUILD_SHARED_LIBS)
    ADD_DEFINITIONS(-DPNG_BUILD_DLL)
  ELSE(BUILD_SHARED_LIBS)
    ADD_DEFINITIONS(-DPNG_STATIC)
  ENDIF(BUILD_SHARED_LIBS)
ENDIF (CYGWIN)
