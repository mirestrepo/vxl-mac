        IF(PYTHON_LIBRARIES)
          TARGET_LINK_LIBRARIES( bbgm_batch optimized ${PYTHON_LIBRARY})
        ENDIF(PYTHON_LIBRARIES)

        IF( BUILD_TESTING )
          SUBDIRS(tests)
        ENDIF( BUILD_TESTING )

      ENDIF(GEOTIFF_FOUND )
